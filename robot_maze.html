def choose_action(epsilon):
    option = [1,0]
    mov = random.choices(option,[(1-epsilon),epsilon])
    if mov==[0]: # 
        action = random.choice(actions) #        
    else: 
        action = max(qline.items(), key=lambda x:x[1])[0] #  Q 
    return action
